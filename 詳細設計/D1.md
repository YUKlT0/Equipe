-- 例: 7日以上前のカート削除
DELETE FROM cart
WHERE created_at < CURRENT_DATE - INTERVAL '7 days';
-- 関連する cart_item も削除


-- 30日以上前に登録された非会員で、注文実績がない顧客を削除
DELETE FROM customer
WHERE membership = false
  AND user_date < CURRENT_DATE - INTERVAL '30 days'
  AND user_id NOT IN (
    SELECT DISTINCT user_id FROM orders
  );
